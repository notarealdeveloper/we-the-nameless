\Chapter{1}

\Verse{1}
{
	\hP{}
}
{
	\eP{
		And He called to Moses, and YHWH spoke to him from the Tent of Meeting,
saying,\\
	}
}
{
}

\Verse{2}
{
	\hP{}
}
{
	\eP{
		“Speak to the children of Israel. And you shall say to them: A human
from you who will make an offering to YHWH—you shall make your offering
from the domestic animals: from the herd and from the flock.\\
	}
}
{
}

\Verse{3}
{
	\hP{}
}
{
	\eP{
		“If his offering is a burnt offering from the herd, he shall make it an
unblemished male. He shall bring it forward to the entrance of the Tent
of Meeting for his acceptance in front of YHWH.\\
	}
}
{
}

\Verse{4}
{
	\hP{}
}
{
	\eP{
		And he shall lay his hand on the head of the burnt offering, and it
will be accepted for him, to atone for him.\\
	}
}
{
}

\Verse{5}
{
	\hP{}
}
{
	\eP{
		And he shall slaughter the herd animal in front of YHWH, and the sons
of Aaron, the priests, shall bring the blood forward and fling the blood
on the altar, all around, which is at the entrance of the Tent of
Meeting.\\
	}
}
{
}

\Verse{6}
{
	\hP{}
}
{
	\eP{
		And he shall flay the burnt offering and cut it into its parts.\\
	}
}
{
}

\Verse{7}
{
	\hP{}
}
{
	\eP{
		And the sons of Aaron, the priest, shall put fire on the altar and
arrange wood on the fire.\\
	}
}
{
}

\Verse{8}
{
	\hP{}
}
{
	\eP{
		And the sons of Aaron, the priests, shall arrange the parts, the head,
and the suet on the wood that is on the fire that is on the altar.\\
	}
}
{
}

\Verse{9}
{
	\hP{}
}
{
	\eP{
		And he shall wash its innards and its legs with water. And the priest
shall burn it all to smoke at the altar, a burnt offering, an offering
by fire of a pleasant smell to YHWH.\\
	}
}
{
}

\Verse{10}
{
	\hP{}
}
{
	\eP{
		“And if his offering is from the flock—from the sheep or from the
goats—for a burnt offering, he shall make it an unblemished male.\\
	}
}
{
}

\Verse{11}
{
	\hP{}
}
{
	\eP{
		And he shall slaughter it on the northward side of the altar in front
of YHWH. And the sons of Aaron, the priests, shall fling its blood on
the altar, all around.\\
	}
}
{
}

\Verse{12}
{
	\hP{}
}
{
	\eP{
		And he shall cut it into its parts and its head and its suet. And the
priest shall arrange them on the wood that is on the fire that is on the
altar.\\
	}
}
{
}

\Verse{13}
{
	\hP{}
}
{
	\eP{
		And he shall wash the innards and the legs with water. And the priest
shall bring it all forward and burn it to smoke at the altar. It is a
burnt offering, an offering by fire of a pleasant smell to YHWH.\\
	}
}
{
}

\Verse{14}
{
	\hP{}
}
{
	\eP{
		“And if his offering to YHWH is a burnt offering from birds, then he
shall make his offering from turtledoves or from pigeons.\\
	}
}
{
}

\Verse{15}
{
	\hP{}
}
{
	\eP{
		And the priest shall bring it forward to the altar and wring its head
and burn it to smoke at the altar, and its blood shall be drained on the
wall of the altar.\\
	}
}
{
}

\Verse{16}
{
	\hP{}
}
{
	\eP{
		And he shall take away its crop with its feathers and throw it beside
the altar eastward to the place of the ashes.\\
	}
}
{
}

\Verse{17}
{
	\hP{}
}
{
	\eP{
		And he shall tear it open by its wings—he shall not divide it—and the
priest shall burn it to smoke at the altar on the wood that is on the
fire. It is a burnt offering, an offering by fire of a pleasant smell to
YHWH.\\
	}
}
{
}
