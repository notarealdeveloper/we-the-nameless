\begin{quote}
\textbf{\hAJ{חנך}} (\textit{ḥ-n-k / ḥănōḵ}) \emph{n.}\
From \hAJ{חנך} (\textit{ḥ-n-k}), meaning \emph{to dedicate}, \emph{to educate}, \emph{to train}, or \emph{to teach}; used as the name \emph{Enoch}.

\medskip

\textbf{\hAJ{הלך}} (\textit{h-l-k / hālak}) \emph{v.}\
From \hAJ{הלך} (\textit{h-l-k}), meaning \emph{to go}, \emph{to walk}, or \emph{to proceed}; movement along a path.

\medskip

\textbf{\hAJ{לקח}} (\textit{l-q-ḥ / lāqaḥ}) \emph{v., n.}\
From \hAJ{לקח} (\textit{l-q-ḥ}), meaning \emph{to take}; also \emph{lesson}, \emph{instruction}, or \emph{education} (\textit{léqaḥ}).
Related to Arabic laḥiqa (“to stick together”) through metathesis.

\medskip

\textbf{\hAJ{לָקַח חֵלֶק}} (\textit{l-q-ḥ ḥ-l-q / lāqaḥ ḥéleq}) \emph{v.}\
From \hAJ{לָקַח} (“to take”) + \hAJ{חֵלֶק} (“part,” “portion”); literally \emph{“to take part.”}; thus \emph{to participate}, \emph{to be a part of}.
\end{quote}
