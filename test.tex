% ------------------------------------------------------------
% Source style primitives
%
% These are intentionally composable.
% Example:
%   \eJ{Hi \cR{there} you}

\newcommand{\estyleJ}{\EnglishJFont\color{JColor}\mdseries}
\newcommand{\estyleP}{\EnglishPFont\color{PColor}\bfseries}
\newcommand{\estyleR}{\EnglishRFont\color{RColor}\mdseries}

\newcommand{\hstyleJ}{\HebrewJFont\color{JColor}\mdseries}
\newcommand{\hstyleP}{\HebrewPFont\color{PColor}\bfseries}
\newcommand{\hstyleR}{\HebrewRFont\color{RColor}\mdseries}

% ------------------------------------------------------------
% Highlighted R spans
%
% lua-ul + luacolor lets the background belong to the text span
% rather than to a paragraph box.
% ------------------------------------------------------------

\newcommand{\eJ}[1]{{\estyleJ #1}}
\newcommand{\eP}[1]{{\estyleP #1}}
\newcommand{\eR}[1]{{\estyleR\highLight[RBg]{#1}}}

\newcommand{\hJ}[1]{{\hstyleJ #1}}
\newcommand{\hP}[1]{{\hstyleP #1}}
\newcommand{\hR}[1]{{\hstyleR\highLight[RBg]{#1}}}

% ------------------------------------------------------------
% Side containers
%
% Each side gets its own independent line breaking.
% Hebrew side owns one outer RTL context.
% ------------------------------------------------------------

\newcommand{\EnglishSide}[1]{%
    \begin{minipage}[t]{\VerseColWidth}
        \EnglishFont
        \raggedright
        #1\par
    \end{minipage}%
}

\newcommand{\HebrewSide}[1]{%
    \begin{minipage}[t]{\VerseColWidth}
        \begin{RTL}
            \HebrewFont
            \raggedleft
            #1\par
        \end{RTL}
    \end{minipage}%
}

% ------------------------------------------------------------
% Book / chapter helpers
% ------------------------------------------------------------

\newcommand{\Book}[1]{%
    {\Huge\bfseries #1\par}%
    \vspace{1.2em}%
}

\newcommand{\Chapter}[1]{%
    {\LARGE\bfseries #1\par}%
    \vspace{1.2em}%
}

% ------------------------------------------------------------
% Verse header
% ------------------------------------------------------------

\newcommand{\VerseHeader}[1]{%
    \par
    \vspace{0.8em}%
    \hrule
    \vspace{0.25em}%
    {\centering\Large #1\par}%
    \vspace{0.25em}%
    \hrule
    \vspace{0.5em}%
}

% ------------------------------------------------------------
% Verse
%
% #1 = verse label
% #2 = English side
% #3 = Hebrew side
% ------------------------------------------------------------

\newcommand{\Verse}[3]{%
    \VerseHeader{#1}%
    \noindent
    \EnglishSide{#2}%
    \hspace{\VerseGap}%
    \HebrewSide{#3}%
    \par
}

% ------------------------------------------------------------
% Commentary helpers
% ------------------------------------------------------------

\newcommand{\asA}[1]{%
    \par
    \vspace{0.35em}%
    {\raggedright #1\par}%
}

\newcommand{\asB}[1]{%
    \par
    \vspace{0.2em}%
    {\raggedright\cR{#1}\par}%
}

\newcommand{\asC}[1]{%
    \par
    \vspace{0.2em}%
    {\raggedright\cB{#1}\par}%
}
