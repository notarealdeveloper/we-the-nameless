\Chapter{8}

\Verse{1}
{
	\hP{ויזכר אלהים את נח ואת כל החיה ואת כל הבהמה אשר אתו בתבה ויעבר אלהים רוח על הארץ וישכו המים}
}
{
	\eP{
		And God remembered Noah\\
		and all the wild animals\\
		and all the domestic animals that were with him in the ark,\\
		and God passed a wind over the earth,\\
		and the water decreased.\\
	}
}
{
}

\Verse{2}
{
	\hP{ויסכרו מעינת תהום וארבת השמים}
	\hJ{ויכלא הגשם מן השמים}
}
{
	\eP{
		And the fountains of the deep\\
		and the apertures of the skies were shut,\\
	}
	\eJ{
		and the rain was restrained from the skies.\\
	}
}
{
}

\Verse{3}
{
	\hJ{וישבו המים מעל הארץ הלוך ושוב}
	\hP{ויחסרו המים מקצה חמשים ומאת יום}
}
{
	\eJ{
		And the waters went back from on the earth,\\
		going back continually,\\
	}
	\eP{
		and the water receded at the end of a hundred fifty days.\\
	}
}
{
}

\Verse{4}
{
	\hP{ותנח התבה בחדש השביעי בשבעה עשר יום לחדש על הרי אררט}
}
{
	\eP{
		And the ark rested in the seventh month,\\
		in the seventeenth day of the month,\\
		on the mountains of Ararat.\\
	}
}
{
}

\Verse{5}
{
	\hP{והמים היו הלוך וחסור עד החדש העשירי בעשירי באחד לחדש נראו ראשי ההרים}
}
{
	\eP{
		And the water went on receding until the tenth month.\\
		In the tenth month, in the first of the month,\\
		the tops of the mountains appeared.\\
	}
}
{
}

\Verse{6}
{
	\hJ{ויהי מקץ ארבעים יום ויפתח נח את חלון התבה אשר עשה}
}
{
	\eJ{
		And it was at the end of forty days,\\
		and Noah opened the window of the ark that he had made.\\
	}
}
{
}

\Verse{7}
{
	\hP{וישלח את הערב ויצא יצוא ושוב עד יבשת המים מעל הארץ}
}
{
	\eP{
		And he let a raven go,\\
		and it went back and forth\\
		until the water dried up from the earth.\\
	}
}
{
}

\Verse{8}
{
	\hJ{וישלח את היונה מאתו לראות הקלו המים מעל פני האדמה}
}
{
	\eJ{
		And he let a dove go from him\\
		to see whether the waters had eased from the face of the earth.\\
	}
}
{
}

\Verse{9}
{
	\hJ{ולא מצאה היונה מנוח לכף רגלה ותשב אליו אל התבה כי מים על פני כל הארץ וישלח ידו ויקחה ויבא אתה אליו אל התבה}
}
{
	\eJ{
		And the dove did not find a resting place for its foot,\\
		and it came back to him to the ark,\\
		for waters were on the face of the earth,\\
		and he put out his hand and took it\\
		and brought it to him to the ark.\\
	}
}
{
}

\Verse{10}
{
	\hJ{ויחל עוד שבעת ימים אחרים ויסף שלח את היונה מן התבה}
}
{
	\eJ{
		And he waited still another seven days,\\
		and he again let a dove go from the ark.\\
	}
}
{
}

\Verse{11}
{
	\hJ{ותבא אליו היונה לעת ערב והנה עלה זית טרף בפיה וידע נח כי קלו המים מעל הארץ}
}
{
	\eJ{
		And the dove came to him at evening time,\\
		and here was an olive leaf torn off in its mouth,\\
		and Noah knew that the waters had eased from the earth.\\
	}
}
{
}

\Verse{12}
{
	\hJ{ויחל עוד שבעת ימים אחרים וישלח את היונה ולא יספה שוב אליו עוד}
}
{
	\eJ{
		And he waited still another seven days,\\
		and he let a dove go,\\
		and it did not come back to him ever again.\\
	}
}
{
}

\Verse{13}
{
	\hP{ויהי באחת ושש מאות שנה בראשון באחד לחדש חרבו המים מעל הארץ}
	\hJ{ויסר נח את מכסה התבה וירא והנה חרבו פני האדמה}
}
{
	\eP{
		And it was in the six hundred and first year,\\
		in the first month, in the first of the month:\\
		the water dried from on the earth.\\
	}
	\eJ{
		And Noah turned back the covering of the ark and looked,\\
		and here the face of the earth had dried.\\
	}
}
{
}

\Verse{14}
{
	\hP{ובחדש השני בשבעה ועשרים יום לחדש יבשה הארץ}
}
{
	\eP{
		And in the second month,\\
		in the twenty-seventh day of the month,\\
		the earth dried up.\\
	}
}
{
}

\Verse{15}
{
	\hP{וידבר אלהים אל נח לאמר}
}
{
	\eP{
		And God spoke to Noah, saying,\\
	}
}
{
}

\Verse{16}
{
	\hP{צא מן התבה אתה ואשתך ובניך ונשי בניך אתך}
}
{
	\eP{
		“Go out from the ark,\\
		you and your wife\\
		and your sons and your sons’ wives with you.\\
	}
}
{
}

\Verse{17}
{
	\hP{כל החיה אשר אתך מכל בשר בעוף ובבהמה ובכל הרמש הרמש על הארץ הוצא אתך ושרצו בארץ ופרו ורבו על הארץ}
}
{
	\eP{
		Bring out with you all the living things that are with you,\\
		of all flesh,\\
		of the birds and of the domestic animals\\
		and of all the creeping animals that creep on the earth,\\
		and they will swarm in the earth\\
		and be fruitful and multiply on the earth.”\\
	}
}
{
}

\Verse{18}
{
	\hP{ויצא נח ובניו ואשתו ונשי בניו אתו}
}
{
	\eP{
		And Noah went out,\\
		and his sons and his wife\\
		and his sons’ wives with him.\\
	}
}
{
}

\Verse{19}
{
	\hP{כל החיה כל הרמש וכל העוף כל רומש על הארץ למשפחתיהם יצאו מן התבה}
}
{
	\eP{
		All the living things,\\
		all the creeping animals and all the birds,\\
		all that creep on the earth\\
		went out from the ark by their families.\\
	}
}
{
}

\Verse{20}
{
	\hJ{ויבן נח מזבח ליהוה ויקח מכל הבהמה הטהורה ומכל העוף הטהור ויעל עלת במזבח}
}
{
	\eJ{
		And Noah built an altar to YHWH,\\
		and he took some of each of the pure animals\\
		and of each of the pure birds,\\
		and he offered sacrifices on the altar.\\
	}
}
{
}

\Verse{21}
{
	\hJ{וירח יהוה את ריח הניחח ויאמר יהוה אל לבו לא אסף לקלל עוד את האדמה בעבור האדם כי יצר לב האדם רע מנעוריו ולא אסף עוד להכות את כל חי כאשר עשיתי}
}
{
	\eJ{
		And YHWH smelled the pleasant smell,\\
		and YHWH said to His heart,\\
		“I won’t curse the ground on account of humankind again,\\
		because the inclination of the human heart is bad from their youth,\\
		and I won’t strike all the living again as I have done.\\
	}
}
{
}

\Verse{22}
{
	\hJ{עד כל ימי הארץ זרע וקציר וקר וחם וקיץ וחרף ויום ולילה לא ישבתו}
}
{
	\eJ{
		All the rest of the earth’s days,\\
		seed and harvest,\\
		and cold and heat,\\
		and summer and winter,\\
		and day and night\\
		will not cease.”\\
	}
}
{
}
