\newcommand{\wetitlepage}{%
    \clearpage
    \newgeometry{left=2.13cm,right=2.13cm,top=2.13cm,bottom=2.13cm}
    \pagestyle{empty}
    {
    \centering
    \vspace*{0.2\textheight}
    {\fontsize{100}{120}\fontfamily{lmr}\selectfont\textsc{We}\par}
    \vspace*{0.2\textheight}
    {\fontsize{25}{30}\fontfamily{lmr}\selectfont\textrm{The Nameless}\par}
    \vspace*{\fill}
    \newpage
    }
    \restoregeometry
    \pagestyle{fancy}
}

\def\englishsize{\normalsize}
\def\hebrewsize{\normalsize}

\newcommand{\VerseEnglish}[1]{%
    \begin{minipage}[t]{\VerseColWidth}
        \EnglishFont
        \englishsize
        \raggedright
        #1\par
    \end{minipage}%
}

\newcommand{\VerseHebrew}[1]{%
    \hbox to \VerseColWidth{%
        \hss
        \vtop{%
            \hsize=\VerseColWidth
            \begin{RTL}
            \HebrewFont
            \hebrewsize
            #1\par
            \end{RTL}
        }%
    }%
}

\newcommand{\VerseColumns}[2]{%
    \noindent
    \VerseEnglish{#1}%
    \hspace{\VerseGap}%
    \vrule width \VerseRuleWidth%
    \hspace{\VerseGap}%
    \VerseHebrew{#2}%
    \par
}

\newcommand{\Verse}[4]{%
    \par
    \vspace{0.8em}%
    \hrule
    \vspace{0.25em}%
    {\centering \large \CurrentBook\ \CurrentChapter:#1\par}%
    \vspace{0.4em}%
    \hrule
    \vspace{0.3em}%
    \begingroup
    \hbadness=10000
    \hfuzz=2em
    \VerseColumns{#3}{#2}%
    \endgroup
    \vspace{0.6em}%
    #4\par
    \vspace{0.8em}%
}

% Commentary / utterance layer

% Generic form:
%   \as{<style macro>}[l|c|r]{text}
% ------------------------------------------------------------

\def\commentarysize{\normalsize}

\NewDocumentCommand{\as}{ m O{l} +m }{%
    \par
    \begingroup
    \parindent=0pt
    \parskip=0pt
    \commentarysize
    \rmfamily
    \def\tempa{#2}%
    \def\tempb{c}%
    \def\tempc{r}%
    \ifx\tempa\tempb
        \centering
    \else
        \ifx\tempa\tempc
            \raggedleft
        \else
            \raggedright
        \fi
    \fi
    #1{#3}\par
    \endgroup
}
