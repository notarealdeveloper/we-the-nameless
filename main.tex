    \endgroup

    #4\par
    \vspace{0.3em}%
}

% ------------------------------------------------------------
% Commentary / utterance layer
%
% Generic form:
%   \as{<style macro>}[l|c|r]{text}
%
% Examples:
%   \as{\cK}{Hello}
%   \as{\cR}[c]{Centered line}
%   \as{\sJ}[r]{J speaks}
% ------------------------------------------------------------

\def\commentarysize{\normalsize}

\def\as@align@l{\raggedright}
\def\as@align@c{\centering}
\def\as@align@r{\raggedleft}

\RenewDocumentCommand{\as}{ m O{l} +m }{%
    \par
    \begingroup
    \parindent=0pt
    \parskip=0pt
    \commentarysize
    \rmfamily
    \def\tempa{#2}%
    \def\tempb{c}%
    \def\tempc{r}%
    \ifx\tempa\tempb
        \centering
    \else
        \ifx\tempa\tempc
            \raggedleft
        \else
            \raggedright
        \fi
    \fi
    #1{#3}\par
    \endgroup
}
% ------------------------------------------------------------
% Speaker shortcuts
%
% A / B / C are arbitrary speakers:
%   A = black
%   B = red
%   C = blue
%
% Source-author shortcuts are also available:
%   J, E, P, R, RJE, Records, Other
% ------------------------------------------------------------

\NewDocumentCommand{\asA}{ O{l} +m }{\as{\cK}[#1]{#2}}
\NewDocumentCommand{\asB}{ O{l} +m }{\as{\cR}[#1]{#2}}
\NewDocumentCommand{\asC}{ O{l} +m }{\as{\cB}[#1]{#2}}

\NewDocumentCommand{\asJ}{ O{l} +m }{\as{\sJ}[#1]{#2}}
\NewDocumentCommand{\asE}{ O{l} +m }{\as{\sE}[#1]{#2}}
\NewDocumentCommand{\asP}{ O{l} +m }{\as{\sP}[#1]{#2}}
\NewDocumentCommand{\asR}{ O{l} +m }{\as{\sR}[#1]{#2}}
\NewDocumentCommand{\asRJE}{ O{l} +m }{\as{\sRJE}[#1]{#2}}
\NewDocumentCommand{\asRecords}{ O{l} +m }{\as{\sBookOfRecords}[#1]{#2}}
\NewDocumentCommand{\asOther}{ O{l} +m }{\as{\sOther}[#1]{#2}}

% ------------------------------------------------------------

\newcommand{\DeclareSpeakerTriplet}[1]{%
    \expandafter\newcommand\csname as#1l\endcsname[1]{\csname as#1\endcsname[l]{##1}}%
    \expandafter\newcommand\csname as#1c\endcsname[1]{\csname as#1\endcsname[c]{##1}}%
    \expandafter\newcommand\csname as#1r\endcsname[1]{\csname as#1\endcsname[r]{##1}}%
}
