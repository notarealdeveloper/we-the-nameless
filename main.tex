\newcommand{\wetitlepage}{%
    \clearpage
    \newgeometry{left=2.13cm,right=2.13cm,top=2.13cm,bottom=2.13cm}
    \pagestyle{empty}
    {
    \centering
    \vspace*{0.2\textheight}
    {\fontsize{100}{120}\fontfamily{lmr}\selectfont\textsc{We}\par}
    \vspace*{0.2\textheight}
    {\fontsize{25}{30}\fontfamily{lmr}\selectfont\textrm{The Nameless}\par}
    \vspace*{\fill}
    \newpage
    }
    \restoregeometry
    \pagestyle{fancy}
}

\def\englishsize{\normalsize}
\def\hebrewsize{\normalsize}

\newcommand{\VerseEnglish}[1]{%
    \begin{minipage}[t]{\VerseColWidth}
        \EnglishFont
        \englishsize
        \raggedright
        #1\par
    \end{minipage}%
}

\newcommand{\VerseHebrew}[1]{%
    \hbox to \VerseColWidth{%
        \hss
        \vtop{%
            \hsize=\VerseColWidth
            \begin{RTL}
            \HebrewFont
            \hebrewsize
            #1\par
            \end{RTL}
        }%
    }%
}

\newcommand{\VerseColumns}[2]{%
    \noindent
    \VerseEnglish{#1}%
    \hspace{\VerseGap}%
    \vrule width \VerseRuleWidth%
    \hspace{\VerseGap}%
    \VerseHebrew{#2}%
    \par
}

\newcommand{\VerseHeader}[1]{%
    \Needspace{6\baselineskip}
    \vspace{0.8em}%
    \hrule
    \vspace{0.25em}%
    {\centering \large \CurrentBook\ \CurrentChapter:#1\par}%
    \vspace{0.8em}%
    \hrule
    \vspace{0.3em}%
}

\newcommand{\Verse}[4]{%
    \par
    \VerseHeader{#1}%
    \begingroup
    \hbadness=10000
    \hfuzz=2em
    \VerseColumns{#3}{#2}%
    \endgroup
    \vspace{0.8em}%
    #4\par
    \vspace{0.8em}%
}
