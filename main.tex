\newcommand{\wetitlepage}{%
    \clearpage
    \newgeometry{left=2.13cm,right=2.13cm,top=2.13cm,bottom=2.13cm}
    \pagestyle{empty}
    {
    \centering
    \vspace*{0.2\textheight}
    {\fontsize{100}{120}\fontfamily{lmr}\selectfont\textsc{We}\par}
    \vspace*{0.2\textheight}
    {\fontsize{25}{30}\fontfamily{lmr}\selectfont\textrm{The Nameless}\par}
    \vspace*{\fill}
    \newpage
    }
    \restoregeometry
    \pagestyle{fancy}
}

\def\englishsize{\normalsize}
\def\hebrewsize{\normalsize}

\newcolumntype{L}{>{\RaggedRight\arraybackslash\hspace{0pt}}p{0.48\linewidth}}
\newcolumntype{R}{>{\beginR\RaggedLeft\arraybackslash\hspace{0pt}}p{0.48\linewidth}<{\endR}}

\newcommand{\VerseEnglish}[1]{%
    \englishsize
    \strut #1%
}

\newcommand{\VerseHebrew}[1]{%
    \hebrewsize
    \strut
    \begingroup
    \BeginHebrewSourceContext
    \heb{#1}%
    \endgroup
}

\newcommand{\Verse}[4]{%
    \par
    \vspace{0.8em}%
    \hrule
    \vspace{0.25em}%
    {\centering \large \CurrentBook\ \CurrentChapter:#1\par}%
    \vspace{0.4em}%
    \hrule
    \vspace{0.3em}%
    \begingroup
    \hbadness=10000
    \hfuzz=2em
    \noindent
    \begin{tabular}{@{}L!{\vrule width 0.2pt}R@{}}
        \VerseEnglish{#3} &
        \VerseHebrew{#2}
    \end{tabular}%
    \par
    \endgroup
    \vspace{0.3em}%
    #4\par
    \vspace{0.3em}%
}
