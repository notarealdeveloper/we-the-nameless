\newcommand{\VerseBlock}[4]{%
    \par
    \vspace{0.9em}%
    \hrule
    \vspace{0.35em}%
    {\centering \large \CurrentBook\ \CurrentChapter:#1\par}%
    \vspace{0.70em}%
    \hrule
    \vspace{0.35em}%
    \noindent
    \begin{tabularx}{\linewidth}{@{}L!{\vrule width 0.2pt}R@{}}%
        {\englishsize #3} & {\hebrewfont\hebrewsize #2}%
    \end{tabularx}%
    \par
    \vspace{0.35em}%
    {#4}%
    \vspace{0.35em}%
}
